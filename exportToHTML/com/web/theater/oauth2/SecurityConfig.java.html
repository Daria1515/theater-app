3    import org.springframework.context.annotation.Bean;
4    import org.springframework.context.annotation.Configuration;
5    import org.springframework.security.config.annotation.web.builders.HttpSecurity;

10   @Configuration
11   @EnableWebSecurity
12   public class SecurityConfig {
13   
14   	@Bean
15   	public SecurityFilterChain securityFilterChain(HttpSecurity http) throws Exception {
16   		http
17   				// Authorize requests
18   				.authorizeHttpRequests((requests) -> requests
19   						.requestMatchers("/**", "/GetAvatarUserController").permitAll()//ко всем страницам разрешить доступ
20   				)
21   				.csrf(AbstractHttpConfigurer::disable)//отключаем проверку csrf для корректных ajax запросов
22   				// Configure OAuth2 login
23   				.oauth2Login(oauth2Login ->
24   						oauth2Login
25   								.loginPage("/")
26   								.defaultSuccessUrl("/reg_sig_google", true)//куда (контроллер) будет получен ответ от google в виде данных пользователя
27   				);
28   		return http.build();
29   	}
